#include <stdio.h>

int main() {
    char op;
    float a, b;

    printf("Enter first number: ");
    scanf("%f", &a);

    printf("Enter operator (+ - * /): ");
    scanf(" %c", &op);

    printf("Enter second number: ");
    scanf("%f", &b);

    switch(op) {
        case '+': printf("Result: %f\n", a + b); break;
        case '-': printf("Result: %f\n", a - b); break;
        case '*': printf("Result: %f\n", a * b); break;
        
        default: printf("Invalid operator!\n");
    }

    return 0;
}
